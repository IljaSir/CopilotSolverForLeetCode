        string res = "";
        int n = title.size();
        for (int i = 0; i < n; i++) {
            if (title[i] == ' ') {
                res += ' ';
            } else {
                int j = i;
                while (j < n && title[j] != ' ') {
                    j++;
                }
                if (j - i <= 2) {
                    for (int k = i; k < j; k++) {
                        res += tolower(title[k]);
                    }
                } else {
                    res += toupper(title[i]);
                    for (int k = i + 1; k < j; k++) {
                        res += tolower(title[k]);
                    }
                }
                i = j - 1;
            }
        }
        return res;

                }

        int len = title.length();

        int i = 0;
        while (i < len) {
            if (title[i] == ' ') {
                ans += ' ';
                i++;
            } else {
                int j = i;
                while (j < len && title[j] != ' ') {
                    j++;
                }
                if (j - i == 1 || j - i == 2) {
                    for (int k = i; k < j; k++) {
                        ans += tolower(title[k]);
                    }
                } else {
                    ans += toupper(title[i]);
                    for (int k = i + 1; k < j; k++) {
                        ans += tolower(title[k]);
                    }
                }
                i = j;
            }
        }
        return ans;
